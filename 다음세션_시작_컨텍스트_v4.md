## 6. 점검 명령

```bash
# 서버 상태
curl -s http://localhost:5001/api/health | python3 -m json.tool

# 자동화 토글 5종
curl -s http://localhost:5001/api/settings | python3 -c "import json,sys;d=json.load(sys.stdin);[print(k,'=',v) for k,v in d.items() if 'auto' in k or 'hubnet' in k]"

# DB 헬스체크
sqlite3 price_history.db "SELECT COUNT(*) FROM model_category; SELECT COUNT(*) FROM bid_cost; SELECT COUNT(*) FROM price_adjustments WHERE status='pending'"

# 최근 자동 조정 이력 (Step 17-C 컬럼 검증)
curl -s "http://localhost:5001/api/auto-adjust/history?limit=3" | python3 -m json.tool

# 식货 활성 batch 분포
sqlite3 price_history.db "SELECT category, COUNT(*) FROM model_category GROUP BY category"

# JQ4110 ONE SIZE 잔여 (자연 소멸 모니터링)
sqlite3 price_history.db "SELECT order_id, model, size, status FROM price_adjustments WHERE model='JQ4110'"

# 서버 재시작
lsof -ti:5001 | xargs kill -9 2>/dev/null
nohup python3 kream_server.py > server.log 2>&1 & disown
sleep 3 && curl -s http://localhost:5001/api/health
```
