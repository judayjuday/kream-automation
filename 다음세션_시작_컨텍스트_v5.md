## 6. 점검 명령

```bash
# 서버 상태
curl -s http://localhost:5001/api/health | python3 -m json.tool

# 자동화 토글 5종
curl -s http://localhost:5001/api/settings | python3 -c "import json,sys;d=json.load(sys.stdin);[print(k,'=',v) for k,v in d.items() if 'auto' in k or 'hubnet' in k]"

# DB 헬스체크
sqlite3 price_history.db "SELECT 'model_category', COUNT(*) FROM model_category UNION ALL SELECT 'pa_pending', COUNT(*) FROM price_adjustments WHERE status='pending' UNION ALL SELECT 'pa_deficit', COUNT(*) FROM price_adjustments WHERE status='deficit' UNION ALL SELECT 'sales_history', COUNT(*) FROM sales_history"

# JQ4110 ONE SIZE 자연 소멸 추적 (status 변화 감지)
sqlite3 price_history.db "SELECT order_id, status, COUNT(*) FROM price_adjustments WHERE model='JQ4110' AND size='ONE SIZE' GROUP BY order_id, status ORDER BY order_id, status"

# my-bids 활성 (KREAM에서 만료되었는지 직접 확인)
curl -s -X POST http://localhost:5001/api/my-bids/sync  # task 폴링
curl -s http://localhost:5001/api/my-bids/local | python3 -c "import json,sys; d=json.load(sys.stdin); jq=[b for b in d.get('bids',[]) if b.get('model')=='JQ4110']; print(f'JQ4110 활성: {len(jq)}건'); [print(f'  {b[\"orderId\"]} size={b[\"size\"]} price={b[\"price\"]:,} rank={b[\"rank\"]}') for b in jq]"

# 세션 mtime
for f in auth_state.json auth_state_kream.json auth_state_hubnet.json; do
  mtime=$(stat -f%m "$f"); diff_hr=$(echo "scale=1; ($(date +%s) - $mtime) / 3600" | bc)
  echo "  📄 $f — ${diff_hr}h"
done

# 서버 재시작
lsof -ti:5001 | xargs kill -9 2>/dev/null
nohup python3 kream_server.py > server.log 2>&1 & disown
sleep 3 && curl -s http://localhost:5001/api/health
```
